42      /**
43       * Copy the project specified into a temporary testing directory. Create the {@link MavenProject} and
44       * {@link ManifestUpdateMojo}, configure it from the <code>plugin-config.xml</code> and return the created Mojo.
45       * <p>
46       * Note: only configuration entries supplied in the plugin-config.xml are presently configured in the mojo returned.
47       * That means and 'default-value' settings are not automatically injected by this testing framework (or plexus
48       * underneath that is suppling this functionality)
49       * 
50       * @param resourceProject
51       *            the name of the goal to look for in the <code>plugin-config.xml</code> that the configuration will be
52       *            pulled from.
53       * @param resourceProject
54       *            the resourceProject path (in src/test/resources) to find the example/test project.
55       * @return the created mojo (unexecuted)
56       * @throws Exception
57       *             if there was a problem creating the mojo.
58       */
59      protected T createMojo(String resourceProject) throws Exception {
60          // Establish test details project example
61          String testResourcePath = "src/test/resources/" + resourceProject;
62          testResourcePath = FilenameUtils.separatorsToSystem(testResourcePath);
63          File exampleDir = new File(getBasedir(), testResourcePath);
64          Assert.assertTrue("Path should exist: " + exampleDir, exampleDir.exists());
65  
66          // Establish the temporary testing directory.
67          String testingPath = "target/tests/" + this.getClass().getSimpleName() + "." + getName();
68          testingPath = FilenameUtils.separatorsToSystem(testingPath);
69          File testingDir = new File(getBasedir(), testingPath);
70  
71          if (testingDir.exists()) {
72              FileUtils.cleanDirectory(testingDir);
73          } else {
74              Assert.assertTrue("Could not create directory: " + testingDir, testingDir.mkdirs());
75          }
76  
77          // Copy project example into temporary testing directory
78          // to avoid messing up the good source copy, as mojo can change
79          // the AndroidManifest.xml file.
80          FileUtils.copyDirectory(exampleDir, testingDir);
81  
82          // Prepare MavenProject
83          final MavenProject project = new MojoProjectStub(testingDir);
84  
85          // Setup Mojo
86          PlexusConfiguration config = extractPluginConfiguration("android-maven-plugin", project.getFile());
87          @SuppressWarnings("unchecked")
88          final T mojo = (T) lookupMojo(getPluginGoalName(), project.getFile());
89  
90          // Inject project itself
91          setVariableValueToObject(mojo, "project", project);
92  
93          // Configure the rest of the pieces via the PluginParameterExpressionEvaluator
94          //  - used for ${plugin.*}
95          MojoDescriptor mojoDesc = new MojoDescriptor();
96          // - used for error messages in PluginParameterExpressionEvaluator
97          mojoDesc.setGoal(getPluginGoalName());
98          MojoExecution mojoExec = new MojoExecution(mojoDesc);
99          // - Only needed if we start to use expressions like ${settings.*}, ${localRepository}, ${reactorProjects}
100         // MavenSession context = null; // Messy to declare, would rather avoid using it.
101         // - Used for ${basedir} relative paths
102         PathTranslator pathTranslator = new DefaultPathTranslator();
103         // - Declared to prevent NPE from logging events in maven core
104         Logger logger = new ConsoleLogger(Logger.LEVEL_DEBUG, mojo.getClass().getName());
105 
106          MavenSession context = createMock(MavenSession.class);
107 
108         expect(context.getExecutionProperties()).andReturn(project.getProperties());
109         expect(context.getCurrentProject()).andReturn(project);
110         replay(context);
111 
112         // Declare evalator that maven itself uses.
113         ExpressionEvaluator evaluator = new PluginParameterExpressionEvaluator(
114                 context, mojoExec, pathTranslator, logger, project, project.getProperties());
115         // Lookup plexus configuration component
116         ComponentConfigurator configurator = (ComponentConfigurator) lookup(ComponentConfigurator.ROLE,"basic");
117         // Configure mojo using above
118         ConfigurationListener listener = new DebugConfigurationListener( logger );
119         configurator.configureComponent( mojo, config, evaluator, getContainer().getContainerRealm(), listener );
120 
121         return mojo;
122     }
