122      * Creates an Android Makefile based on the specified set of static library dependency artifacts.
123      *
124      * @param outputDir         Directory to resolve artifact locations relative to.  Makefiles contain relative paths
125      * @param artifacts         The list of (static library) dependency artifacts to create the Makefile from
126      * @param useHeaderArchives If true, the Makefile should include a LOCAL_EXPORT_C_INCLUDES statement, pointing to
127      *                          the location where the header archive was expanded
128      * @param repoSession
129      * @param projectRepos
130      * @param repoSystem
131      * @return The created Makefile
132      */
133     public MakefileHolder createMakefileFromArtifacts( File outputDir, Set<Artifact> artifacts,
134                                                               String ndkArchitecture, String defaultNDKArchitecture,
135                                                               boolean useHeaderArchives )
136             throws IOException, MojoExecutionException
137     {
138 
139         final StringBuilder makeFile = new StringBuilder( "# Generated by Android Maven Plugin\n" );
140         final List<File> includeDirectories = new ArrayList<File>();
141 
142         // Add now output - allows us to somewhat intelligently determine the include paths to use for the header
143         // archive
144         makeFile.append( "$(shell echo \"LOCAL_C_INCLUDES=$(LOCAL_C_INCLUDES)\" > $(" + MAKEFILE_CAPTURE_FILE + "))" );
145         makeFile.append( '\n' );
146         makeFile.append( "$(shell echo \"LOCAL_PATH=$(LOCAL_PATH)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
147         makeFile.append( '\n' );
148         makeFile.append( "$(shell echo \"LOCAL_MODULE_FILENAME=$(LOCAL_MODULE_FILENAME)\" >> $("
149                 + MAKEFILE_CAPTURE_FILE + "))" );
150         makeFile.append( '\n' );
151         makeFile.append( "$(shell echo \"LOCAL_MODULE=$(LOCAL_MODULE)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
152         makeFile.append( '\n' );
153         makeFile.append( "$(shell echo \"LOCAL_CFLAGS=$(LOCAL_CFLAGS)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
154         makeFile.append( '\n' );
155 
156         if ( ! artifacts.isEmpty() )
157         {
158             for ( Artifact artifact : artifacts )
159             {
160                 final String architecture = NativeHelper.extractArchitectureFromArtifact( artifact,
161                         defaultNDKArchitecture );
162 
163                 makeFile.append( '\n' );
164                 makeFile.append( "ifeq ($(TARGET_ARCH_ABI)," ).append( architecture ).append( ")\n" );
165 
166                 makeFile.append( "#\n" );
167                 makeFile.append( "# Group ID: " );
168                 makeFile.append( artifact.getGroupId() );
169                 makeFile.append( '\n' );
170                 makeFile.append( "# Artifact ID: " );
171                 makeFile.append( artifact.getArtifactId() );
172                 makeFile.append( '\n' );
173                 makeFile.append( "# Artifact Type: " );
174                 makeFile.append( artifact.getType() );
175                 makeFile.append( '\n' );
176                 makeFile.append( "# Version: " );
177                 makeFile.append( artifact.getVersion() );
178                 makeFile.append( '\n' );
179                 makeFile.append( "include $(CLEAR_VARS)" );
180                 makeFile.append( '\n' );
181                 makeFile.append( "LOCAL_MODULE    := " );
182                 makeFile.append( artifact.getArtifactId() );
183                 makeFile.append( '\n' );
184 
185                 final boolean apklibStatic = addLibraryDetails( makeFile, outputDir, artifact, ndkArchitecture );
186 
187                 if ( useHeaderArchives )
188                 {
189                     try
190                     {
191                         Artifact harArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
192                                 artifact.getVersion(), artifact.getScope(), "har", artifact.getClassifier(),
193                                 artifact.getArtifactHandler() );
194                         final Artifact resolvedHarArtifact = AetherHelper
195                                 .resolveArtifact( harArtifact, repoSystem, repoSession, projectRepos );
196 
197                         File includeDir = new File( System.getProperty( "java.io.tmpdir" ),
198                                 "android_maven_plugin_native_includes" + System.currentTimeMillis() + "_"
199                                         + resolvedHarArtifact.getArtifactId() );
200                         includeDir.deleteOnExit();
201                         includeDirectories.add( includeDir );
202 
203                         JarHelper.unjar( new JarFile( resolvedHarArtifact.getFile() ), includeDir,
204                                 new JarHelper.UnjarListener()
205                                 {
206                                     @Override
207                                     public boolean include( JarEntry jarEntry )
208                                     {
209                                         return ! jarEntry.getName().startsWith( "META-INF" );
210                                     }
211                                 } );
212 
213                         makeFile.append( "LOCAL_EXPORT_C_INCLUDES := " );
214                         final String str = includeDir.getAbsolutePath();
215                         makeFile.append( str );
216                         makeFile.append( '\n' );
217                         
218                         if ( log.isDebugEnabled() )
219                         {
220                             Collection<File> includes = FileUtils.listFiles( includeDir,
221                                     TrueFileFilter.INSTANCE, TrueFileFilter.INSTANCE );
222                             log.debug( "Listing LOCAL_EXPORT_C_INCLUDES for " + artifact.getId() + ": " + includes );
223                         }
224                     }
225                     catch ( Exception e )
226                     {
227                         throw new MojoExecutionException(
228                                 "Error while resolving header archive file for: " + artifact.getArtifactId(), e );
229                     }
230                 }
231                 if ( "a".equals( artifact.getType() ) || apklibStatic )
232                 {
233                     makeFile.append( "include $(PREBUILT_STATIC_LIBRARY)\n" );
234                 }
235                 else
236                 {
237                     makeFile.append( "include $(PREBUILT_SHARED_LIBRARY)\n" );
238                 }
239 
240                 makeFile.append( "endif #" ).append( artifact.getClassifier() ).append( '\n' );
241                 makeFile.append( '\n' );
242 
243             }
244         }
245         
246         return new MakefileHolder( includeDirectories, makeFile.toString() );
247     }

249     private boolean addLibraryDetails( StringBuilder makeFile, File outputDir,
250                                        Artifact artifact, String ndkArchitecture ) throws IOException
251     {
252         boolean apklibStatic = false;
253         if ( AndroidExtension.APKLIB.equals( artifact.getType() ) )
254         {
255             String classifier = artifact.getClassifier();
256             String architecture = ( classifier != null ) ? classifier : ndkArchitecture;
257             // 
258             // We assume that APKLIB contains a single static OR shared library
259             // that we should link against. The follow code identifies that file.
260             //
261             File[] staticLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
262                                                               architecture, true );
263             if ( staticLibs != null && staticLibs.length > 0 )
264             {
265                 int libIdx = findApklibNativeLibrary( staticLibs, artifact.getArtifactId() );
266                 apklibStatic = true;
267                 addLibraryDetails( makeFile, outputDir, staticLibs[libIdx], "" );
268             }
269             else
270             {
271                 File[] sharedLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
272                                                                   architecture, false );
273                 if ( sharedLibs == null )
274                 {
275                     throw new IOException( "Failed to find any library file in APKLIB" );
276                 }
277                 int libIdx = findApklibNativeLibrary( sharedLibs, artifact.getArtifactId() );
278                 addLibraryDetails( makeFile, outputDir, sharedLibs[libIdx], "" );
279             }
280         }
281         else
282         {
283             addLibraryDetails( makeFile, outputDir, artifact.getFile(), artifact.getArtifactId() );
284         }
285 
286         return apklibStatic;
287     }

289     private void addLibraryDetails( StringBuilder makeFile, File outputDir, File libFile, String outputName )
290         throws IOException
291     {
292         String localPath = resolveRelativePath( outputDir, libFile );
293         localPath = localPath.substring( 0, localPath.indexOf( libFile.getName() ) - 1 );
294 
295         makeFile.append( "LOCAL_PATH := " );
296         makeFile.append( localPath );
297         makeFile.append( '\n' );
298         makeFile.append( "LOCAL_SRC_FILES := " );
299         makeFile.append( libFile.getName() );
300         makeFile.append( '\n' );
301         makeFile.append( "LOCAL_MODULE_FILENAME := " );
302         if ( "".equals( outputName ) )
303         {
304             makeFile.append( FilenameUtils.removeExtension( libFile.getName() ) );
305         }
306         else
307         {
308             makeFile.append( outputName );
309         }
310         makeFile.append( '\n' );
311     }

314      * @param libs the array of possible library files. Must not be null.
315      * @return the index in the array of the library to use
316      * @throws IOException if a library cannot be identified
317      */
318     private int findApklibNativeLibrary( File[] libs, String artifactName ) throws IOException
319     {
320         int libIdx = -1;
321         
322         if ( libs.length == 1 )
323         
324         {
325             libIdx = 0;
326         }
327         else
328         {
329             log.info( "Found multiple library files, looking for name match with artifact" );
330             StringBuilder sb = new StringBuilder();
331             for ( int i = 0; i < libs.length; i++ )
332             {
333                 if ( sb.length() != 0 )
334                 {
335                     sb.append( ", " );
336                 }
337                 sb.append( libs[i].getName() );
338                 if ( libs[i].getName().startsWith( "lib" + artifactName ) )
339                 {
340                     if ( libIdx != -1 )
341                     {
342                         // We have multiple matches, tell the user we can't handle this ...
343                         throw new IOException( "Found multiple libraries matching artifact name " + artifactName
344                                 + ". Please use unique artifact/library names." );
345                         
346                     }
347                     libIdx = i;
348                 }
349             }
350             if ( libIdx < 0 )
351             {
352                 throw new IOException( "Unable to determine main library from " + sb.toString()
353                         + " APKLIB should contain only 1 library or a library matching the artifact name" );
354             }
355         }
356         return libIdx;
357     }

359     /**
360      * Resolves the relative path of the specified artifact
361      *
362      * @param outputDirectory typically the parent directory of the directory containing the makefile
363      * @param file
364      * @return
365      */
366     protected static String resolveRelativePath( File outputDirectory, File file ) throws IOException
367     {
368         String resolvedPath = file.getCanonicalPath();
369         
370         String strOutputDirectoryPath = outputDirectory.getCanonicalPath();
371         String strFilePath = file.getCanonicalPath();
372         //System.out.println( "Resolving " + strFilePath + " against " + strOutputDirectoryPath );
373 
374         if ( strFilePath.startsWith( strOutputDirectoryPath ) )
375         {
376             // Simple case where file is in a subdirectory of outputDirectory
377             resolvedPath =  strFilePath.substring( strOutputDirectoryPath.length() + 1 );
378         }
379         else
380         {
381             // Look for commonality in paths
382             List<String> outputDirectoryPathParts = splitPath( outputDirectory.getCanonicalFile() );
383             List<String> filePathParts = splitPath( file.getCanonicalFile() );
384             int commonDepth = 0;
385             int maxCommonDepth = Math.min( outputDirectoryPathParts.size(), filePathParts.size() );
386             for ( int i = 0; 
387                     ( i < maxCommonDepth ) 
388                     && outputDirectoryPathParts.get( i ).equals( filePathParts.get( i ) ); 
389                     i++ )
390             {
391                 commonDepth++;
392             }
393             // If there is a common root build a relative path between the common roots
394             if ( commonDepth > 0 )
395             {
396                 final StringBuilder stringBuilder = new StringBuilder();
397                 stringBuilder.append( ".." );
398                 for ( int i = 0; i < outputDirectoryPathParts.size() - commonDepth - 1; i++ )
399                 {
400                     stringBuilder.append( File.separator );
401                     stringBuilder.append( ".." );
402                 }
403                 for ( int i = commonDepth; i < filePathParts.size(); i++ )
404                 {
405                     stringBuilder.append( File.separator );
406                     stringBuilder.append( filePathParts.get( i ) );
407                 }
408                 resolvedPath = stringBuilder.toString();
409             }
410             else
411             {
412                 if ( IS_WINDOWS )
413                 {
414                     // Windows has no common root directory, cannot resolve a path
415                     // across drives so ...
416                     throw new IOException( "Unable to resolve relative path across windows drives" );
417                 }
418 
419                 // no intersection between paths so calculate a path via the root directory
420                 final StringBuilder stringBuilder = new StringBuilder();
421                 
422                 File depthCheck = outputDirectory.getParentFile();
423                 while ( depthCheck != null )
424                 {
425                     if ( stringBuilder.length() > 0 )
426                     {
427                         stringBuilder.append( File.separator );
428                     }
429                     stringBuilder.append( ".." );
430                     depthCheck = depthCheck.getParentFile();
431                 }
432     
433                 resolvedPath = stringBuilder.toString() + strFilePath;
434             }
435         }
436 
437         //System.out.println( "resolvedPath = " + resolvedPath );
438         return resolvedPath;
439     }

441     /**
442      * Method to split the path components of a file into a List
443      * @param f the file to split
444      * @return a new list containing the components of the path as strings
445      */
446     protected static List<String> splitPath( File f )
447     {
448         List<String> result;
449         File parent = f.getParentFile();
450         if ( parent == null )
451         {
452             result = new ArrayList<String>();
453             if ( f.getName().length() > 0 )
454             {
455                 // We're at the root but have a name so we have a relative path
456                 // for which we need to add the first component to the list
457                 result.add( f.getName() );
458             }
459             else if ( IS_WINDOWS ) 
460             {
461                 String strF = f.toString();
462                 if ( strF.matches( WINDOWS_DRIVE_ROOT_REGEX ) )
463                 {
464                     // We're on windows and the path is <Drive>:\ so we
465                     // add the <Drive>: to the list
466                     result.add( strF.substring( 0, strF.length() - 1 ).toUpperCase() );
467                 }
468             }
469         }
470         else
471         {
472             result = splitPath( parent );
473             result.add( f.getName() );
474         }
475         return result;
476     }

478     /**
479      * Creates a list of artifacts suitable for use in the LOCAL_STATIC_LIBRARIES or LOCAL_SHARED_LIBRARIES 
480      * variable in an Android makefile
481      *
482      * @param resolvedLibraryList
483      * @param staticLibrary
484      * @return a list of Ids for artifacts that include static or shared libraries
485      */
486     public String createLibraryList( Set<Artifact> resolvedLibraryList,
487                                      String ndkArchitecture,
488                                      boolean staticLibrary )
489     {
490         Set<String> libraryNames = new LinkedHashSet<String>();
491 
492         for ( Artifact a : resolvedLibraryList )
493         {
494             if ( staticLibrary && "a".equals( a.getType() ) )
495             {
496                 libraryNames.add( a.getArtifactId() );
497             }
498             if ( ! staticLibrary && "so".equals( a.getType() ) )
499             {
500                 libraryNames.add( a.getArtifactId() );
501             }
502             if ( AndroidExtension.APKLIB.equals( a.getType() ) || AndroidExtension.AAR.equals( a.getType() ) )
503             {
504                 File[] libFiles = NativeHelper.listNativeFiles( a, unpackedApkLibsDirectory, 
505                                                                 ndkArchitecture, staticLibrary );
506                 if ( libFiles != null && libFiles.length > 0 )
507                 {
508                     libraryNames.add( a.getArtifactId() );
509                 }
510                 
511             }
512         }
513 
514         StringBuilder sb = new StringBuilder();
515 
516         Iterator<String> iter = libraryNames.iterator();
517 
518         while ( iter.hasNext() )
519         {
520             sb.append( iter.next() );
521 
522             if ( iter.hasNext() )
523             {
524                 sb.append( " " );
525             }
526         }
527 
528         return sb.toString();
529     }
