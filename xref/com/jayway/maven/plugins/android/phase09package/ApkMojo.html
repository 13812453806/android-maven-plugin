182     /**
183      * Pattern for additional META-INF resources to be packaged into the apk.
184      * <p>
185      * The APK builder filters these resources and doesn't include them into the apk.
186      * This leads to bad behaviour of dependent libraries relying on these resources,
187      * for instance service discovery doesn't work.<br/>
188      * By specifying this pattern, the android plugin adds these resources to the final apk.
189      * </p>
190      * <p>The pattern is relative to META-INF, i.e. one must use
191      * <pre>
192      * <code>
193      * &lt;apkMetaIncludes&gt;
194      *     &lt;metaInclude>services/**&lt;/metaInclude&gt;
195      * &lt;/apkMetaIncludes&gt;
196      * </code>
197      * </pre>
198      * ... instead of
199      * <pre>
200      * <code>
201      * &lt;apkMetaIncludes&gt;
202      *     &lt;metaInclude>META-INF/services/**&lt;/metaInclude&gt;
203      * &lt;/apkMetaIncludes&gt;
204      * </code>
205      * </pre>
206      * <p>
207      * See also <a href="http://code.google.com/p/maven-android-plugin/issues/detail?id=97">Issue 97</a>
208      * </p>
209      *
210      * @parameter expression="${android.apk.metaIncludes}" default-value=""
211      */
212     @PullParameter( defaultValueGetterMethod = "getDefaultMetaIncludes" )
213     private String[] apkMetaIncludes;

267     /**
268      *
269      * @throws MojoExecutionException
270      * @throws MojoFailureException
271      */
272     public void execute() throws MojoExecutionException, MojoFailureException
273     {
274 
275         // Make an early exit if we're not supposed to generate the APK
276         if ( ! generateApk )
277         {
278             return;
279         }
280 
281         ConfigHandler cfh = new ConfigHandler( this );
282 
283         cfh.parseConfiguration();
284 
285         generateIntermediateApk();
286 
287         // Compile resource exclusion patterns, if any
288         if ( excludeJarResources != null && excludeJarResources.length > 0 ) 
289         {
290           getLog().debug( "Compiling " + excludeJarResources.length + " patterns" );
291           
292           excludeJarResourcesPatterns = new Pattern[excludeJarResources.length];
293           
294           for ( int index = 0; index < excludeJarResources.length; ++index ) 
295           {
296             excludeJarResourcesPatterns[index] = Pattern.compile( excludeJarResources[index] );
297           }
298         }
299         
300         // Initialize apk build configuration
301         File outputFile = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + "." + APK );
302         final boolean signWithDebugKeyStore = getAndroidSigner().isSignWithDebugKeyStore();
303 
304         if ( getAndroidSigner().shouldCreateBothSignedAndUnsignedApk() )
305         {
306             getLog().info( "Creating debug key signed apk file " + outputFile );
307             createApkFile( outputFile, true );
308             final File unsignedOutputFile = new File( project.getBuild().getDirectory(),
309                     project.getBuild().getFinalName() + "-unsigned." + APK );
310             getLog().info( "Creating additional unsigned apk file " + unsignedOutputFile );
311             createApkFile( unsignedOutputFile, false );
312             projectHelper.attachArtifact( project, unsignedOutputFile,
313                     classifier == null ? "unsigned" : classifier + "_unsigned" );
314         }
315         else
316         {
317             createApkFile( outputFile, signWithDebugKeyStore );
318         }
319 
320         if ( classifier == null )
321         {
322             // Set the generated .apk file as the main artifact (because the pom states <packaging>apk</packaging>)
323             project.getArtifact().setFile( outputFile );
324         }
325         else
326         {
327             // If there is a classifier specified, attach the artifact using that
328             projectHelper.attachArtifact( project, outputFile, classifier );
329         }
330     }

332     void createApkFile( File outputFile, boolean signWithDebugKeyStore ) throws MojoExecutionException
333     {
334         File dexFile = new File( project.getBuild().getDirectory(), "classes.dex" );
335         File zipArchive = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + ".ap_" );
336         ArrayList<File> sourceFolders = new ArrayList<File>();
337         if ( sourceDirectories != null )
338         {
339             for ( File f : sourceDirectories )
340             {
341                 sourceFolders.add( f );
342             }
343         }
344         ArrayList<File> jarFiles = new ArrayList<File>();
345         ArrayList<File> nativeFolders = new ArrayList<File>();
346 
347         // Process the native libraries, looking both in the current build directory as well as
348         // at the dependencies declared in the pom.  Currently, all .so files are automatically included
349         processNativeLibraries( nativeFolders );
350         
351         doAPKWithAPKBuilder( outputFile, dexFile, zipArchive, sourceFolders, jarFiles, nativeFolders,
352                     signWithDebugKeyStore );
353 
354         if ( this.apkMetaIncludes != null && this.apkMetaIncludes.length > 0 )
355         {
356             try
357             {
358                 addMetaInf( outputFile, jarFiles );
359             }
360             catch ( IOException e )
361             {
362                 throw new MojoExecutionException( "Could not add META-INF resources.", e );
363             }
364         }
365     }

367     private void addMetaInf( File outputFile, ArrayList<File> jarFiles ) throws IOException
368     {
369         File tmp = File.createTempFile( outputFile.getName(), ".add", outputFile.getParentFile() );
370 
371         FileOutputStream fos = new FileOutputStream( tmp );
372         ZipOutputStream zos = new ZipOutputStream( fos );
373         Set<String> entries = new HashSet<String>();
374 
375         updateWithMetaInf( zos, outputFile, entries, false );
376 
377         for ( File f : jarFiles )
378         {
379             updateWithMetaInf( zos, f, entries, true );
380         }
381 
382         zos.close();
383 
384         outputFile.delete();
385 
386         if ( ! tmp.renameTo( outputFile ) )
387         {
388             throw new IOException( String.format( "Cannot rename %s to %s", tmp, outputFile.getName() ) );
389         }
390     }

392     private void updateWithMetaInf( ZipOutputStream zos, File jarFile, Set<String> entries, boolean metaInfOnly )
393             throws IOException
394     {
395         ZipFile zin = new ZipFile( jarFile );
396 
397         for ( Enumeration<? extends ZipEntry> en = zin.entries(); en.hasMoreElements(); )
398         {
399             ZipEntry ze = en.nextElement();
400 
401             if ( ze.isDirectory() )
402             {
403                 continue;
404             }
405 
406             String zn = ze.getName();
407 
408             if ( metaInfOnly )
409             {
410                 if ( ! zn.startsWith( "META-INF/" ) )
411                 {
412                     continue;
413                 }
414 
415                 if ( this.extractDuplicates && ! entries.add( zn ) )
416                 {
417                     continue;
418                 }
419 
420                 if ( ! metaInfMatches( zn ) )
421                 {
422                     continue;
423                 }
424             }
425 
426             zos.putNextEntry( new ZipEntry( zn ) );
427 
428             InputStream is = zin.getInputStream( ze );
429 
430             copyStreamWithoutClosing( is, zos );
431 
432             is.close();
433             zos.closeEntry();
434         }
435 
436         zin.close();
437     }

439     private boolean metaInfMatches( String path )
440     {
441         for ( String inc : this.apkMetaIncludes )
442         {
443             if ( SelectorUtils.matchPath( "META-INF/" + inc, path ) )
444             {
445                 return true;
446             }
447         }
448 
449         return false;
450     }

452     private Map<String, List<File>> jars = new HashMap<String, List<File>>();

454     private void computeDuplicateFiles( File jar ) throws IOException
455     {
456         ZipFile file = new ZipFile( jar );
457         Enumeration<? extends ZipEntry> list = file.entries();
458         while ( list.hasMoreElements() )
459         {
460             ZipEntry ze = list.nextElement();
461             if ( ! ( ze.getName().contains( "META-INF/" ) || ze.isDirectory() ) )
462             { // Exclude META-INF and Directories
463                 List<File> l = jars.get( ze.getName() );
464                 if ( l == null )
465                 {
466                     l = new ArrayList<File>();
467                     jars.put( ze.getName(), l );
468                 }
469                 l.add( jar );
470             }
471         }
472     }

474     /**
475      * Creates the APK file using the internal APKBuilder.
476      *
477      * @param outputFile            the output file
478      * @param dexFile               the dex file
479      * @param zipArchive            the classes folder
480      * @param sourceFolders         the resources
481      * @param jarFiles              the embedded java files
482      * @param nativeFolders         the native folders
483      * @param signWithDebugKeyStore enables the signature of the APK using the debug key
484      * @throws MojoExecutionException if the APK cannot be created.
485      */
486     private void doAPKWithAPKBuilder( File outputFile, File dexFile, File zipArchive, ArrayList<File> sourceFolders,
487                                       ArrayList<File> jarFiles, ArrayList<File> nativeFolders,
488                                       boolean signWithDebugKeyStore ) throws MojoExecutionException
489     {
490         getLog().debug( "Building APK with internal APKBuilder" );
491         sourceFolders.add( new File( project.getBuild().getOutputDirectory() ) );
492 
493         for ( Artifact artifact : getRelevantCompileArtifacts() )
494         {
495             if ( extractDuplicates )
496             {
497                 try
498                 {
499                     computeDuplicateFiles( artifact.getFile() );
500                 }
501                 catch ( Exception e )
502                 {
503                     getLog().warn( "Cannot compute duplicates files from " + artifact.getFile().getAbsolutePath(), e );
504                 }
505             }
506             jarFiles.add( artifact.getFile() );
507         }
508 
509         // Check duplicates.
510         if ( extractDuplicates )
511         {
512             List<String> duplicates = new ArrayList<String>();
513             List<File> jarToModify = new ArrayList<File>();
514             for ( String s : jars.keySet() )
515             {
516                 List<File> l = jars.get( s );
517                 if ( l.size() > 1 )
518                 {
519                     getLog().warn( "Duplicate file " + s + " : " + l );
520                     duplicates.add( s );
521                     for ( int i = 1; i < l.size(); i++ )
522                     {
523                         if ( ! jarToModify.contains( l.get( i ) ) )
524                         {
525                             jarToModify.add( l.get( i ) );
526                         }
527                     }
528                 }
529             }
530 
531             // Rebuild jars.
532             for ( File file : jarToModify )
533             {
534                 File newJar;
535                 newJar = removeDuplicatesFromJar( file, duplicates );
536                 int index = jarFiles.indexOf( file );
537                 if ( newJar != null )
538                 {
539                     jarFiles.set( index, newJar );
540                 }
541 
542             }
543         }
544         
545         String debugKeyStore;
546         ApkBuilder apkBuilder; 
547         try 
548         {
549             debugKeyStore = ApkBuilder.getDebugKeystore();
550             apkBuilder = 
551                     new ApkBuilder( outputFile, zipArchive, dexFile, 
552                             ( signWithDebugKeyStore ) ? debugKeyStore : null, null );
553             if ( apkDebug )
554             {
555                 apkBuilder.setDebugMode( apkDebug );
556             }
557 
558             for ( File sourceFolder : sourceFolders )
559             {
560                 apkBuilder.addSourceFolder( sourceFolder );
561             }
562      
563             for ( File jarFile : jarFiles )
564             {
565                 boolean excluded = false;
566               
567                 if ( excludeJarResourcesPatterns != null )
568                 {
569                     final String name = jarFile.getName();
570                     getLog().debug( "Checking " + name + " against patterns" );
571                     for ( Pattern pattern : excludeJarResourcesPatterns )
572                     {
573                         final Matcher matcher = pattern.matcher( name );
574                         if ( matcher.matches() ) 
575                         {
576                             getLog().debug( "Jar " + name + " excluded by pattern " + pattern );
577                             excluded = true;
578                             break;
579                         } 
580                         else 
581                         {
582                             getLog().debug( "Jar " + name + " not excluded by pattern " + pattern );
583                         }
584                     }
585                 }
586     
587                 if ( excluded )
588                 {
589                     continue;
590                 }
591                 
592                 if ( jarFile.isDirectory() )
593                 {
594                     String[] filenames = jarFile.list( new FilenameFilter()
595                     {
596                         public boolean accept( File dir, String name )
597                         {
598                             return PATTERN_JAR_EXT.matcher( name ).matches();
599                         }
600                     } );
601     
602                     for ( String filename : filenames )
603                     {
604                         apkBuilder.addResourcesFromJar( new File( jarFile, filename ) );
605                     }
606                 }
607                 else
608                 {
609                     apkBuilder.addResourcesFromJar( jarFile );
610                 }
611             }
612 
613             for ( File nativeFolder : nativeFolders )
614             {
615                 apkBuilder.addNativeLibraries( nativeFolder );
616             }
617             apkBuilder.sealApk();
618         } 
619         catch ( ApkCreationException e )
620         {
621             throw new MojoExecutionException( e.getMessage() );
622         } 
623         catch ( DuplicateFileException e )
624         {
625             throw new MojoExecutionException( e.getMessage() );
626         } 
627         catch ( SealedApkException e )
628         {
629             throw new MojoExecutionException( e.getMessage() );
630         }
631     }

633     private File removeDuplicatesFromJar( File in, List<String> duplicates )
634     {
635         String target = project.getBuild().getOutputDirectory();
636         File tmp = new File( target, "unpacked-embedded-jars" );
637         tmp.mkdirs();
638         File out = new File( tmp, in.getName() );
639 
640         if ( out.exists() )
641         {
642             return out;
643         }
644         else
645         {
646             try
647             {
648                 out.createNewFile();
649             }
650             catch ( IOException e )
651             {
652                 e.printStackTrace();
653             }
654         }
655 
656         // Create a new Jar file
657         FileOutputStream fos = null;
658         ZipOutputStream jos = null;
659         try
660         {
661             fos = new FileOutputStream( out );
662             jos = new ZipOutputStream( fos );
663         }
664         catch ( FileNotFoundException e1 )
665         {
666             getLog().error( "Cannot remove duplicates : the output file " + out.getAbsolutePath() + " does not found" );
667             return null;
668         }
669 
670         ZipFile inZip = null;
671         try
672         {
673             inZip = new ZipFile( in );
674             Enumeration<? extends ZipEntry> entries = inZip.entries();
675             while ( entries.hasMoreElements() )
676             {
677                 ZipEntry entry = entries.nextElement();
678                 // If the entry is not a duplicate, copy.
679                 if ( ! duplicates.contains( entry.getName() ) )
680                 {
681                     // copy the entry header to jos
682                     jos.putNextEntry( entry );
683                     InputStream currIn = inZip.getInputStream( entry );
684                     copyStreamWithoutClosing( currIn, jos );
685                     currIn.close();
686                     jos.closeEntry();
687                 }
688             }
689         }
690         catch ( IOException e )
691         {
692             getLog().error( "Cannot removing duplicates : " + e.getMessage() );
693             return null;
694         }
695 
696         try
697         {
698             if ( inZip != null )
699             {
700                 inZip.close();
701             }
702             jos.close();
703             fos.close();
704             jos = null;
705             fos = null;
706         }
707         catch ( IOException e )
708         {
709             // ignore it.
710         }
711         getLog().info( in.getName() + " rewritten without duplicates : " + out.getAbsolutePath() );
712         return out;
713     }

715     /**
716      * Copies an input stream into an output stream but does not close the streams.
717      *
718      * @param in  the input stream
719      * @param out the output stream
720      * @throws IOException if the stream cannot be copied
721      */
722     private static void copyStreamWithoutClosing( InputStream in, OutputStream out ) throws IOException
723     {
724         final int bufferSize = 4096;
725         byte[] b = new byte[ bufferSize ];
726         int n;

749     private void processNativeLibraries( final List<File> natives, String ndkArchitecture )
750             throws MojoExecutionException
751     {
752         // Examine the native libraries directory for content. This will only be true if:
753         // a) the directory exists
754         // b) it contains at least 1 file
755         final boolean hasValidNativeLibrariesDirectory = hasValidNativeLibrariesDirectory();
756         final boolean hasValidBuildNativeLibrariesDirectory = hasValidBuildNativeLibrariesDirectory();
757         final Set<Artifact> artifacts = getNativeLibraryArtifacts();
758 
759         if ( artifacts.isEmpty() && hasValidNativeLibrariesDirectory && ! hasValidBuildNativeLibrariesDirectory )
760         {
761 
762             getLog().debug(
763                     "No native library dependencies detected, will point directly to " + nativeLibrariesDirectory );
764 
765             // Point directly to the directory in this case - no need to copy files around
766             addNativeDirectory( natives, nativeLibrariesDirectory );
767 
768             // FIXME: This would pollute a libs folder which is under source control
769             // FIXME: Would be better to not support this case?
770             optionallyCopyGdbServer( nativeLibrariesDirectory, ndkArchitecture );
771 
772         }
773         else
774         {
775             if ( ! artifacts.isEmpty() || hasValidNativeLibrariesDirectory )
776             {
777                 // In this case, we may have both .so files in it's normal location
778                 // as well as .so dependencies
779 
780                 final File destinationDirectory = makeNativeLibrariesOutputDirectory();
781 
782                 // Point directly to the directory
783                 addNativeDirectory( natives, destinationDirectory );
784 
785                 // If we have a valid native libs, copy those files - these already come in the structure required
786                 if ( hasValidNativeLibrariesDirectory )
787                 {
788                     copyLocalNativeLibraries( nativeLibrariesDirectory, destinationDirectory );
789                 }
790 
791                 if ( ! artifacts.isEmpty() )
792                 {
793                     for ( Artifact resolvedArtifact : artifacts )
794                     {
795                         if ( NativeHelper.artifactHasHardwareArchitecture( resolvedArtifact,
796                                 ndkArchitecture, nativeLibrariesDependenciesHardwareArchitectureDefault ) )
797                         {
798                             copyNativeLibraryArtifactFileToDirectory( resolvedArtifact, destinationDirectory,
799                                     ndkArchitecture );
800                         }
801                         else if ( APKLIB.equals( resolvedArtifact.getType() )
802                                 || AAR.equals( resolvedArtifact.getType() ) )
803                         {
804                                 addNativeDirectory( natives, new File( getLibraryUnpackDirectory( resolvedArtifact )
805                                                                            + "/libs" ) );
806                         }
807                     }
808                 }
809 
810                 // Finally, think about copying the gdbserver binary into the APK output as well
811                 optionallyCopyGdbServer( destinationDirectory, ndkArchitecture );
812             }
813         }
814     }

956     /**
957      * Generates an intermediate apk file (actually .ap_) containing the resources and assets.
958      *
959      * @throws MojoExecutionException
960      */
961     private void generateIntermediateApk() throws MojoExecutionException
962     {
963         CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
964         executor.setLogger( this.getLog() );
965         File[] overlayDirectories = getResourceOverlayDirectories();
966 
967         if ( extractedDependenciesRes.exists() )
968         {
969             copyDependenciesRes();
970         }
971         if ( resourceDirectory.exists() && combinedRes.exists() )
972         {
973             copyLocalResourceFiles();
974         }
975 
976         // Must combine assets.
977         // The aapt tools does not support several -A arguments.
978         // We copy the assets from extracted dependencies first, and then the local assets.
979         // This allows redefining the assets in the current project
980         if ( extractedDependenciesAssets.exists() )
981         {
982             copyDependencyAssets();
983         }
984 
985         processApkLibAssets();
986 
987         if ( assetsDirectory.exists() )
988         {
989             copyLocalAssets();
990         }
991 
992         File androidJar = getAndroidSdk().getAndroidJar();
993         File outputFile = new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + ".ap_" );
994 
995         List<String> commands = new ArrayList<String>();
996         commands.add( "package" );
997         commands.add( "-f" );
998         commands.add( "-M" );
999         commands.add( androidManifestFile.getAbsolutePath() );
1000         for ( File resOverlayDir : overlayDirectories )
1001         {
1002             if ( resOverlayDir != null && resOverlayDir.exists() )
1003             {
1004                 commands.add( "-S" );
1005                 commands.add( resOverlayDir.getAbsolutePath() );
1006             }
1007         }
1008         if ( combinedRes.exists() )
1009         {
1010             commands.add( "-S" );
1011             commands.add( combinedRes.getAbsolutePath() );
1012         }
1013         else
1014         {
1015             if ( resourceDirectory.exists() )
1016             {
1017                 commands.add( "-S" );
1018                 commands.add( resourceDirectory.getAbsolutePath() );
1019             }
1020         }
1021         for ( Artifact artifact : getAllRelevantDependencyArtifacts() )
1022         {
1023             if ( artifact.getType().equals( APKLIB ) || artifact.getType().equals( AAR ) )
1024             {
1025                 final String apkLibResDir = getLibraryUnpackDirectory( artifact ) + "/res";
1026                 if ( new File( apkLibResDir ).exists() )
1027                 {
1028                     commands.add( "-S" );
1029                     commands.add( apkLibResDir );
1030                 }
1031             }
1032         }
1033         commands.add( "--auto-add-overlay" );
1034 
1035         // Use the combined assets.
1036         // Indeed, aapt does not support several -A arguments.
1037         if ( combinedAssets.exists() )
1038         {
1039             commands.add( "-A" );
1040             commands.add( combinedAssets.getAbsolutePath() );
1041         }
1042 
1043         if ( StringUtils.isNotBlank( renameManifestPackage ) )
1044         {
1045             commands.add( "--rename-manifest-package" );
1046             commands.add( renameManifestPackage );
1047         }
1048 
1049         if ( StringUtils.isNotBlank( renameInstrumentationTargetPackage ) )
1050         {
1051             commands.add( "--rename-instrumentation-target-package" );
1052             commands.add( renameInstrumentationTargetPackage );
1053         }
1054 
1055         commands.add( "-I" );
1056         commands.add( androidJar.getAbsolutePath() );
1057         commands.add( "-F" );
1058         commands.add( outputFile.getAbsolutePath() );
1059         if ( StringUtils.isNotBlank( configurations ) )
1060         {
1061             commands.add( "-c" );
1062             commands.add( configurations );
1063         }
1064 
1065         for ( String aaptExtraArg : aaptExtraArgs )
1066         {
1067             commands.add( aaptExtraArg );
1068         }
1069 
1070         if ( !release )
1071         {
1072             getLog().info( "Enabling debug build for apk." );
1073             commands.add( "--debug-mode" );
1074         }
1075         else 
1076         {
1077             getLog().info( "Enabling release build for apk." );
1078         }
1079 
1080         getLog().info( getAndroidSdk().getAaptPath() + " " + commands.toString() );
1081         try
1082         {
1083             executor.executeCommand( getAndroidSdk().getAaptPath(), commands, project.getBasedir(), false );
1084         }
1085         catch ( ExecutionException e )
1086         {
1087             throw new MojoExecutionException( "", e );
1088         }
1089     }
