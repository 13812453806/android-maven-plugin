42  /**
43   * Copy file to all the attached (or specified) devices/emulators.
44   *
45   * @author Manfred Moser <manfred@simpligility.com>
46   * @goal push
47   * @requiresProject false
48   */
49  @SuppressWarnings( "unused" )
50  public class PushMojo extends AbstractAndroidMojo
51  {
52  
53      /**
54       * <p>The configuration for the push goal can be set up in the plugin configuration in the pom file as:</p>
55       * <pre>
56       * &lt;push&gt;
57       *     &lt;source&gt;path&lt;/source&gt;
58       *     &lt;destination&gt;path&lt;/destination&gt;
59       * &lt;/push&gt;
60       * </pre>
61       * <p>The parameters can also be configured as property in the pom or settings file
62       * <pre>
63       * &lt;properties&gt;
64       *     &lt;android.push.source&gt;pathondevice&lt;/android.push.source&gt;
65       *     &lt;android.push.destination&gt;path&lt;/android.push.destination&gt;
66       * &lt;/properties&gt;
67       * </pre>
68       * or from command-line with parameter
69       * <code>-Dandroid.push.source=path</code>
70       * and
71       * <code>-Dandroid.push.destination=path</code>.</p>
72       *
73       * @parameter
74       */
75      @ConfigPojo
76      private Push push;
77  
78      /**
79       * The file name of the local filesystem file to push to the emulator or
80       * device either as absolute path or relative to the execution folder.
81       * <p/>
82       * If you specify a directory, all containing files will be pushed recursively.
83       *
84       * @parameter expression="${android.push.source}"
85       */
86      private String pushSource;
87  
88      @PullParameter( required = true )
89      private String parsedSource;
90  
91      /**
92       * The destination file name as absolute path on the emulator or device.
93       * If the last character is a "/" it will be assumed that the original
94       * base filename should be preserved and a target directory is specified.
95       * This works analogous if the source is a directory.
96       *
97       * @parameter expression="${android.push.destination}"
98       */
99      private String pushDestination;
100 
101     @PullParameter( required = true )
102     private String parsedDestination;
103 
104     public void execute() throws MojoExecutionException, MojoFailureException
105     {
106 
107         ConfigHandler configHandler = new ConfigHandler( this, this.session, this.execution );
108         configHandler.parseConfiguration();
109 
110         final Map<String, String> sourceDestinationMap = calculateSourceDestinationMapping();
111 
112         doWithDevices( new DeviceCallback()
113         {
114             public void doWithDevice( final IDevice device ) throws MojoExecutionException
115             {
116                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
117 
118                 // message will be set in for each loop according to the processed files
119                 String message = "";
120 
121                 try
122                 {
123                     SyncService syncService = device.getSyncService();
124 
125                     for ( Map.Entry<String, String> pushFileEntry : sourceDestinationMap.entrySet() )
126                     {
127                         String sourcePath = pushFileEntry.getKey();
128                         String destinationPath = pushFileEntry.getValue();
129 
130                         message = deviceLogLinePrefix + "Push of " + sourcePath + " to " + destinationPath + " on "
131                                 + DeviceHelper.getDescriptiveName( device );
132 
133                         syncService.pushFile( sourcePath, destinationPath, new LogSyncProgressMonitor( getLog() ) );
134 
135                         getLog().info( message + " successful." );
136                     }
137                 }
138                 catch ( SyncException e )
139                 {
140                     throw new MojoExecutionException( message + " failed.", e );
141                 }
142                 catch ( IOException e )
143                 {
144                     throw new MojoExecutionException( message + " failed.", e );
145                 }
146                 catch ( TimeoutException e )
147                 {
148                     throw new MojoExecutionException( message + " failed.", e );
149                 }
150                 catch ( AdbCommandRejectedException e )
151                 {
152                     throw new MojoExecutionException( message + " failed.", e );
153                 }
154             }
155         } );
156     }
157 
158     /**
159      * Calculates a map which contains all files to be pushed to the device or
160      * emulator. The source filename works as the key while the value is the
161      * destination.
162      *
163      * @return a map with file source -> destination pairs
164      * @throws MojoExecutionException
165      */
166     private Map<String, String> calculateSourceDestinationMapping() throws MojoExecutionException
167     {
168         Map<String, String> result = new HashMap<String, String>();
169 
170         File sourceFile = new File( parsedSource );
171         final String destinationPath;
172         if ( parsedDestination.endsWith( "/" ) )
173         {
174             destinationPath = parsedDestination + sourceFile.getName();
175         }
176         else
177         {
178             destinationPath = parsedDestination;
179         }
180 
181         if ( sourceFile.isFile() )
182         {
183             // only put the source in
184             final String sourcePath = sourceFile.getAbsolutePath();
185             result.put( sourcePath, destinationPath );
186         }
187         else
188         {
189             if ( sourceFile.isDirectory() )
190             {
191                 // find recursively all files to be pushed
192                 @SuppressWarnings( "unchecked" ) Collection<File> filesList = FileUtils
193                         .listFiles( sourceFile, null, true );
194                 for ( File file : filesList )
195                 {
196                     // make the file's path relative - this is kind of a hack but it
197                     // works just fine in this controlled environment
198                     String filePath = file.getAbsolutePath().substring( sourceFile.getAbsolutePath().length() );
199                     filePath = filePath.replace( System.getProperty( "file.separator" ), "/" );
200                     result.put( file.getAbsolutePath(), destinationPath + filePath );
201                 }
202             }
203             else
204             {
205                 throw new MojoExecutionException( "Cannot execute push goal: File or directory "
206                         + sourceFile.getAbsolutePath() + " does not exist." );
207             }
208         }
209         return result;
210     }
