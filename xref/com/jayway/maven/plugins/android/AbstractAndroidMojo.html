501     /**
502      *
503      * @param allArtifacts
504      * @return
505      */
506     private Set<Artifact> filterOutIrrelevantArtifacts( Iterable<Artifact> allArtifacts )
507     {
508         final Set<Artifact> results = new LinkedHashSet<Artifact>();
509         for ( Artifact artifact : allArtifacts )
510         {
511             if ( artifact == null )
512             {
513                 continue;
514             }
515 
516             if ( EXCLUDED_DEPENDENCY_SCOPES.contains( artifact.getScope() ) )
517             {
518                 continue;
519             }
520 
521             if ( APK.equalsIgnoreCase( artifact.getType() ) )
522             {
523                 continue;
524             }
525 
526             results.add( artifact );
527         }
528         return results;
529     }

531     /**
532      * Attempts to resolve an {@link Artifact} to a {@link File}.
533      *
534      * @param artifact to resolve
535      * @return a {@link File} to the resolved artifact, never <code>null</code>.
536      * @throws MojoExecutionException if the artifact could not be resolved.
537      */
538     protected File resolveArtifactToFile( Artifact artifact ) throws MojoExecutionException
539     {
540         Artifact resolvedArtifact = AetherHelper.resolveArtifact( artifact, repoSystem, repoSession, projectRepos );
541         final File jar = resolvedArtifact.getFile();
542         if ( jar == null )
543         {
544             throw new MojoExecutionException( "Could not resolve artifact " + artifact.getId()
545                     + ". Please install it with \"mvn install:install-file ...\" or deploy it to a repository "
546                     + "with \"mvn deploy:deploy-file ...\"" );
547         }
548         return jar;
549     }

551     /**
552      * Initialize the Android Debug Bridge and wait for it to start. Does not reinitialize it if it has
553      * already been initialized (that would through and IllegalStateException...). Synchronized sine
554      * the init call in the library is also synchronized .. just in case.
555      *
556      * @return
557      */
558     protected AndroidDebugBridge initAndroidDebugBridge() throws MojoExecutionException
559     {
560         synchronized ( ADB_LOCK )
561         {
562             if ( ! adbInitialized )
563             {
564                 AndroidDebugBridge.init( false );
565                 adbInitialized = true;
566             }
567             AndroidDebugBridge androidDebugBridge = AndroidDebugBridge
568                     .createBridge( getAndroidSdk().getAdbPath(), false );
569             waitUntilConnected( androidDebugBridge );
570             return androidDebugBridge;
571         }
572     }

574     /**
575      * Run a wait loop until adb is connected or trials run out. This method seems to work more reliably then using a
576      * listener.
577      *
578      * @param adb
579      */
580     private void waitUntilConnected( AndroidDebugBridge adb )
581     {
582         int trials = 10;
583         final int connectionWaitTime = 50;
584         while ( trials > 0 )
585         {
586             try
587             {
588                 Thread.sleep( connectionWaitTime );
589             }
590             catch ( InterruptedException e )
591             {
592                 e.printStackTrace();
593             }
594             if ( adb.isConnected() )
595             {
596                 break;
597             }
598             trials--;
599         }
600     }

602     /**
603      * Wait for the Android Debug Bridge to return an initial device list.
604      *
605      * @param androidDebugBridge
606      * @throws MojoExecutionException
607      */
608     protected void waitForInitialDeviceList( final AndroidDebugBridge androidDebugBridge ) throws MojoExecutionException
609     {
610         if ( ! androidDebugBridge.hasInitialDeviceList() )
611         {
612             getLog().info( "Waiting for initial device list from the Android Debug Bridge" );
613             long limitTime = System.currentTimeMillis() + ADB_TIMEOUT_MS;
614             while ( ! androidDebugBridge.hasInitialDeviceList() && ( System.currentTimeMillis() < limitTime ) )
615             {
616                 try
617                 {
618                     Thread.sleep( 1000 );
619                 }
620                 catch ( InterruptedException e )
621                 {
622                     throw new MojoExecutionException(
623                             "Interrupted waiting for initial device list from Android Debug Bridge" );
624                 }
625             }
626             if ( ! androidDebugBridge.hasInitialDeviceList() )
627             {
628                 getLog().error( "Did not receive initial device list from the Android Debug Bridge." );
629             }
630         }
631     }

633     /**
634      * Deploys an apk file to a connected emulator or usb device.
635      *
636      * @param apkFile the file to deploy
637      * @throws MojoExecutionException If there is a problem deploying the apk file.
638      */
639     protected void deployApk( final File apkFile ) throws MojoExecutionException, MojoFailureException
640     {
641         if ( undeployBeforeDeploy )
642         {
643             undeployApk( apkFile );
644         }
645         doWithDevices( new DeviceCallback()
646         {
647             public void doWithDevice( final IDevice device ) throws MojoExecutionException
648             {
649                 String deviceLogLinePrefix = DeviceHelper.getDeviceLogLinePrefix( device );
650                 try
651                 {
652                     String result = device.installPackage( apkFile.getAbsolutePath(), true );
653                     // according to the docs for installPackage, not null response is error
654                     if ( result != null )
655                     {
656                         throw new MojoExecutionException( deviceLogLinePrefix 
657                                 + "Install of " + apkFile.getAbsolutePath()
658                                 + " failed - [" + result + "]" );
659                     }
660                     getLog().info( deviceLogLinePrefix + "Successfully installed " + apkFile.getAbsolutePath() + " to "
661                             + DeviceHelper.getDescriptiveName( device ) );
662                 }
663                 catch ( InstallException e )
664                 {
665                     throw new MojoExecutionException( deviceLogLinePrefix + "Install of " + apkFile.getAbsolutePath() 
666                             + " failed.", e );
667                 }
668             }
669         } );
670     }

672     /**
673      *
674      * @throws MojoExecutionException
675      * @throws MojoFailureException
676      */
677     protected void deployDependencies() throws MojoExecutionException, MojoFailureException
678     {
679         Set<Artifact> directDependentArtifacts = project.getDependencyArtifacts();
680         if ( directDependentArtifacts != null )
681         {
682             for ( Artifact artifact : directDependentArtifacts )
683             {
684                 String type = artifact.getType();
685                 if ( type.equals( APK ) )
686                 {
687                     getLog().debug( "Detected apk dependency " + artifact + ". Will resolve and deploy to device..." );
688                     final File targetApkFile = resolveArtifactToFile( artifact );
689                     if ( undeployBeforeDeploy )
690                     {
691                         getLog().debug( "Attempting undeploy of " + targetApkFile + " from device..." );
692                         undeployApk( targetApkFile );
693                     }

732     {
733         final AndroidDebugBridge androidDebugBridge = initAndroidDebugBridge();
734

738         }
739

741         List<IDevice> devices = Arrays.asList( androidDebugBridge.getDevices() );
742         int numberOfDevices = devices.size();

747         }
748

822      *

830     private boolean shouldDoWithThisDevice( IDevice idevice ) throws MojoExecutionException, MojoFailureException
831     {
832         // use specified device or all emulators or all devices
833         if ( "emulator".equals( device ) && idevice.isEmulator() )
834         {
835             return true;
836         }
837 
838         if ( "usb".equals( device ) && ! idevice.isEmulator() )
839         {
840             return true;
841         }
842 
843         if ( idevice.isEmulator() && ( device.equalsIgnoreCase( idevice.getAvdName() ) || device
844                 .equalsIgnoreCase( idevice.getSerialNumber() ) ) )
845         {
846             return true;
847         }
848 
849         if ( ! idevice.isEmulator() && device.equals( idevice.getSerialNumber() ) )
850         {
851             return true;
852         }
853 
854         return false;
855     }

946      * Extracts the package name from an XmlTree dump of AndroidManifest.xml by the <code>aapt</code> tool.
947      *
948      * @param aaptDumpXmlTree output from <code>aapt dump xmltree &lt;apkFile&gt; AndroidManifest.xml
949      * @return the package name from inside the apkFile.
950      */
951     protected String extractPackageNameFromAndroidManifestXmlTree( String aaptDumpXmlTree )
952     {
953         final Scanner scanner = new Scanner( aaptDumpXmlTree );
954         // Finds the root element named "manifest".
955         scanner.findWithinHorizon( "^E: manifest", 0 );
956         // Finds the manifest element's attribute named "package".
957         scanner.findWithinHorizon( "  A: package=\"", 0 );
958         // Extracts the package value including the trailing double quote.
959         String packageName = scanner.next( ".*?\"" );
960         // Removes the double quote.
961         packageName = packageName.substring( 0, packageName.length() - 1 );
962         return packageName;
