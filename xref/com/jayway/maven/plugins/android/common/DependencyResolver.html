19  /**
20   * Resolves the aar and apklib dependencies for an Artifact.
21   *
22   * @author William Ferguson <william.ferguson@xandar.com.au>
23   */
24  public final class DependencyResolver
25  {
26      private final Logger log;
27      private final DependencyGraphBuilder dependencyGraphBuilder;
28  
29      public DependencyResolver( Logger log, DependencyGraphBuilder dependencyGraphBuilder )
30      {
31          this.log = log;
32          this.dependencyGraphBuilder = dependencyGraphBuilder;
33      }
34  
35      /**
36       * @param project   MavenProject for which to return the dependencies.
37       * @param session   MavenSession in which to look for reactor dependencies.
38       * @return all the dependencies for a project.
39       * @throws DependencyGraphBuilderException if the dependency graph can't be built.
40       */
41      public Set<Artifact> getProjectDependenciesFor( MavenProject project, MavenSession session )
42              throws DependencyGraphBuilderException
43      {
44          // No need to filter our search. We want to resolve all artifacts.
45          final DependencyNode node = dependencyGraphBuilder.buildDependencyGraph( project, null, session.getProjects() );
46  
47          final DependencyCollector collector = new DependencyCollector( log, project.getArtifact() );
48          collector.visit( node, false );
49          return collector.getDependencies();
50      }
51  
52      /**
53       * Returns the Set of APKLIB, AAR, APK (direct or transitive) dependencies of the supplied artifact.
54       *
55       * The project is searched until artifact is found and then the library dependencies are looked for recursively.
56       *
57       * @param project   MavenProject that contains the artifact.
58       * @param artifact  Artifact for whom to get the dependencies.
59       * @return Set of APK, APKLIB and AAR dependencies.
60       * @throws org.apache.maven.plugin.MojoExecutionException if it couldn't resolve any of the dependencies.
61       */
62      public Set<Artifact> getLibraryDependenciesFor( MavenProject project, final Artifact artifact )
63              throws MojoExecutionException
64      {
65          // Set a filter that should only return interesting artifacts.
66          final ArtifactFilter filter = new ArtifactFilter()
67          {
68              @Override
69              public boolean include( Artifact found )
70              {
71                  final String type = found.getType();
72                  return ( type.equals( APKLIB ) || type.equals( AAR ) || type.equals( APK ) );
73              }
74          };
75  
76          final DependencyNode node;
77          try
78          {
79              node = dependencyGraphBuilder.buildDependencyGraph( project, filter );
80          }
81          catch ( DependencyGraphBuilderException e )
82          {
83              throw new MojoExecutionException( "Could not resolve project dependency graph", e );
84          }
85  
86          final DependencyCollector collector = new DependencyCollector( log, artifact );
87          collector.visit( node, false );
88          return collector.getDependencies();
89      }
90  }
