37  /**
38   * Adds classes from AAR and APK dependencies to the project compile classpath.
39   */
40  @Component( role = AbstractMavenLifecycleParticipant.class, hint = "default" )
41  public final class ClasspathModifierLifecycleParticipant extends AbstractMavenLifecycleParticipant
42  {
43      @SuppressWarnings( "unused" )
44      @Requirement
45      private ArtifactResolver artifactResolver;
46  
47      @SuppressWarnings( "unused" )
48      @Requirement( hint = "default" )
49      private DependencyGraphBuilder dependencyGraphBuilder;
50  
51      @SuppressWarnings( "unused" )
52      @Requirement
53      private Logger log;
54  
55      @Override
56      public void afterProjectsRead( MavenSession session ) throws MavenExecutionException
57      {
58          log.debug( "" );
59          log.debug( "ClasspathModifierLifecycleParticipant#afterProjectsRead - start" );
60          log.debug( "" );
61  
62          log.debug( "CurrentProject=" + session.getCurrentProject() );
63          final List<MavenProject> projects = session.getProjects();
64          final DependencyResolver dependencyResolver = new DependencyResolver( log, dependencyGraphBuilder );
65          final ArtifactResolverHelper artifactResolverHelper = new ArtifactResolverHelper( artifactResolver, log,
66                  Collections.<ArtifactRepository>emptyList() );
67  
68          for ( MavenProject project : projects )
69          {
70              log.debug( "" );
71              log.debug( "project=" + project.getArtifact() );
72  
73              final UnpackedLibHelper helper = new UnpackedLibHelper( artifactResolverHelper, project, log );
74  
75              final Set<Artifact> artifacts;
76  
77              // If there is an extension ClassRealm loaded for this project then use that
78              // as the ContextClassLoader so that Wagon extensions can be used to resolves dependencies.
79              final ClassLoader projectClassLoader = ( project.getClassRealm() != null )
80                      ? project.getClassRealm()
81                      : Thread.currentThread().getContextClassLoader();
82  
83              final ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
84              try
85              {
86                  Thread.currentThread().setContextClassLoader( projectClassLoader );
87                  artifacts = dependencyResolver.getProjectDependenciesFor( project, session );
88              }
89              catch ( DependencyGraphBuilderException e )
90              {
91                  // Nothing to do. The resolution failure will be displayed by the standard resolution mechanism.
92                  continue;
93              }
94              finally
95              {
96                  Thread.currentThread().setContextClassLoader( originalClassLoader );
97              }
98  
99              log.debug( "projects deps: : " + artifacts );
100             for ( Artifact artifact : artifacts )
101             {
102                 final String type = artifact.getType();
103                 if ( type.equals( AndroidExtension.AAR ) )
104                 {
105                     // An AAR lib contains a classes jar that needs to be added to the classpath.
106                     // Create a placeholder classes.jar and add it to the compile classpath.
107                     // It will replaced with the real classes.jar by GenerateSourcesMojo.
108                     addClassesToClasspath( helper, project, artifact );
109                 }
110                 else if ( type.equals( AndroidExtension.APK ) )
111                 {
112                     // The only time that an APK will likely be a dependency is when this an an APK test project.
113                     // So add a placeholder (we cannot resolve the actual dep pre build) to the compile classpath.
114                     // The placeholder will be replaced with the real APK jar later.
115                     addClassesToClasspath( helper, project, artifact );
116                 }
117             }
118         }
119         log.debug( "" );
120         log.debug( "ClasspathModifierLifecycleParticipant#afterProjectsRead - finish" );
121     }
122 
123     /**
124      * Add the dependent library classes to the project classpath.
125      */
126     private void addClassesToClasspath( UnpackedLibHelper helper, MavenProject project, Artifact artifact )
127         throws MavenExecutionException
128     {
129         // Work out where the dep will be extracted and calculate the file path to the classes jar.
130         // This is location where the GenerateSourcesMojo will extract the classes.
131         final File classesJar = helper.getUnpackedClassesJar( artifact );
132         log.debug( "Adding to classpath : " + classesJar );
133 
134         // In order to satisfy the LifecycleDependencyResolver on execution up to a phase that
135         // has a Mojo requiring dependency resolution I need to create a dummy classesJar here.
136         classesJar.getParentFile().mkdirs();
137         try
138         {
139             classesJar.createNewFile();
140             log.debug( "Created dummy " + classesJar.getName() + " exist=" + classesJar.exists() );
141         }
142         catch ( IOException e )
143         {
144             throw new MavenExecutionException( "Could not add " + classesJar.getName() + " as dependency", e );
145         }
146 
147         // Add the classes to the classpath
148         final Dependency dependency = createSystemScopeDependency( artifact, classesJar );
149         project.getModel().addDependency( dependency );
150     }
151 
152     private Dependency createSystemScopeDependency( Artifact artifact, File location )
153     {
154         final Dependency dependency = new Dependency();
155         dependency.setGroupId( artifact.getGroupId() );
156         dependency.setArtifactId( artifact.getArtifactId() );
157         dependency.setVersion( artifact.getVersion() );
158         dependency.setScope( Artifact.SCOPE_SYSTEM );
159         dependency.setSystemPath( location.getAbsolutePath() );
160         return dependency;
161     }
162 }
